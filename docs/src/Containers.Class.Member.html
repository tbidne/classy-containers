-- | Types that have a notion of member.
--
-- @since 0.1
class Member a where
  type MElem a

  -- | @since 0.1
  member :: MElem a -> a -> Bool

-- | @since 0.1
instance Eq a => Member [a] where
  type MElem [a] = a
  member :: MElem [a] -> [a] -> Bool
member = forall (t :: * -> *) a. (Foldable t, Eq a) => a -> t a -> Bool
elem

-- | @since 0.1
instance HashCompat k => Member (HashMap k v) where
  type MElem (HashMap k v) = k
  member :: MElem (HashMap k v) -> HashMap k v -> Bool
member = forall k a. (Eq k, Hashable k) => k -> HashMap k a -> Bool
HMap.member

-- | @since 0.1
instance HashCompat a => Member (HashSet a) where
  type MElem (HashSet a) = a
  member :: MElem (HashSet a) -> HashSet a -> Bool
member = forall a. (Eq a, Hashable a) => a -> HashSet a -> Bool
HSet.member

-- | @since 0.1
instance Member (IntMap a) where
