-- | Types that support an "ordered" insert.
--
-- @since 0.1
class Sequenced a where
  -- | Element type.
  type SElem a

  -- | Appends an element to the sequence.
  --
  -- @since 0.1
  append :: a -> SElem a -> a

  -- | Prepends an element to the sequence.
  --
  -- @since 0.1
  prepend :: SElem a -> a -> a

-- | Operator alias for 'append'. U+22D7.
--
-- @since 0.1
