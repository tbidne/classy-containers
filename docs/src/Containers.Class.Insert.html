-- | Types that support an "unordered" insert.
--
-- @since 0.1
class Insert a where
  -- | Constraint on the element.
  type InsertC a :: Constraint

  -- | Element type.
  type IElem a

  -- Inserts an element.
  --
  -- @since 0.1
  insert :: InsertC a => IElem a -> a -> a
